function compress(str) {
  let compressed = "";
  arr = str.split("");
  let obj = {};
  for (i = 0; i < arr.length; i++) {
    if (obj[arr[i]]) {
      obj[arr[i]]++;
    } else {
      obj[arr[i]] = 1;
    }
  }
  for (let key in obj) {
    compressed = compressed + key + obj[key];
  }
  console.log(compressed);
}
